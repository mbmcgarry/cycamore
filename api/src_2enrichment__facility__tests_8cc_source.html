00519   using cyclus::CompMap;
00520   using cyclus::Composition;
00521   using cyclus::Material;
00522   using cyclus::Request;
00523   using cyclus::Trade;
00524   using cyclus::toolkit::MatQuery;
00525   using cyclus::toolkit::Assays;
00526   using cyclus::toolkit::FeedQty;
00527   using cyclus::toolkit::SwuRequired;
00528   using cyclus::toolkit::UraniumAssay;
00529   using test_helpers::get_mat;
00530 
00531   // problem set up
00532   std::vector< cyclus::Trade<cyclus::Material> > trades;
00533   std::vector<std::pair<cyclus::Trade<cyclus::Material>,
00534                         cyclus::Material::Ptr> > responses;
00535 
00536   double qty = 5;  // 5 kg
00537   double trade_qty = qty / 3;
00538   double product_assay = 0.05;  // of 5 w/o enriched U
00539   cyclus::CompMap v;
00540   v[922350000] = product_assay;
00541   v[922380000] = 1 - product_assay;
00542   // target qty need not be = to request qty

00544       qty + 10, cyclus::Composition::CreateFromMass(v));

00546   Assays assays(feed_assay, UraniumAssay(target), tails_assay);
00547   double swu_req = SwuRequired(qty, assays);

00550   src_facility->SetMaxInventorySize(natu_req * 4);  // not capacitated by nat u
00551   src_facility->SwuCapacity(swu_req);  // swu capacitated
00552 
00553   // Null response
00554   src_facility->GetMatlTrades(trades, responses);
00555   EXPECT_NO_THROW();
00556   EXPECT_EQ(responses.size(), 0);
00557 
00558   // set up state
00559   DoAddMat(GetMat(natu_req * 2));
00560 
00561   Request<Material>* req =
00562       Request<Material>::Create(target, trader, out_commod);
