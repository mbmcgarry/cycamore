00706 std::set<cyclus::RequestPortfolio<cyclus::Material>::Ptr>
00707 BatchReactor::GetMatlRequests() {
00708   using cyclus::RequestPortfolio;
00709   using cyclus::Material;
00710 
00711   std::set<RequestPortfolio<Material>::Ptr> set;
00712   double order_size;
00713 
00714   switch (phase()) {
00715     // the initial phase requests as much fuel as necessary to achieve an entire
00716     // core
00717     case INITIAL:
00718       order_size = n_batches() * batch_size()
00719                    - core_.quantity() - reserves_.quantity()
00720                    - spillover_->quantity();
00721       if (preorder_time() == 0) {
00722         order_size += batch_size() * n_reserves();
00723       }
00724       if (order_size > 0) {
00725         RequestPortfolio<Material>::Ptr p = GetOrder_(order_size);
00726         set.insert(p);
00727       }
00728       break;
00729 
00730     // the default case is to request the reserve amount if the order time has
00731     // been reached
00732     default:
00733       // double fuel_need = (n_reserves() + n_batches() - n_core()) * batch_size();
00734       double fuel_need = (n_reserves() + n_load()) * batch_size();
00735       double fuel_have = reserves_.quantity() + spillover_->quantity();
00736       order_size = fuel_need - fuel_have;
00737       bool ordering = order_time() <= context()->time() && order_size > 0;
00738 
00739       LOG(cyclus::LEV_DEBUG5, "BReact") << "BatchReactor " << prototype()
00740                                         << " is deciding whether to order -";
00741       LOG(cyclus::LEV_DEBUG5, "BReact") << "    Needs fuel amt: " << fuel_need;
00742       LOG(cyclus::LEV_DEBUG5, "BReact") << "    Has fuel amt: " << fuel_have;
00743       LOG(cyclus::LEV_DEBUG5, "BReact") << "    Order amt: " << order_size;
00744       LOG(cyclus::LEV_DEBUG5, "BReact") << "    Order time: " << order_time();
00745       LOG(cyclus::LEV_DEBUG5, "BReact") << "    Current time: "
00746                                         << context()->time();
00747       LOG(cyclus::LEV_DEBUG5, "BReact") << "    Ordering?: "
00748                                         << ((ordering == true) ? "yes" : "no");
00749 
00750       if (ordering) {
00751         RequestPortfolio<Material>::Ptr p = GetOrder_(order_size);
00752         set.insert(p);
00753       }
00754       break;
00755   }
00756 
00757   return set;
00758 }

00944 // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00945 void BatchReactor::AddBatches_(std::string commod, cyclus::Material::Ptr mat) {
00946   using cyclus::Material;
00947   using cyclus::ResCast;
00948 
00949   LOG(cyclus::LEV_DEBUG3, "BReact") << "BatchReactor " << prototype()
00950                                     << " is adding " << mat->quantity()
00951                                     << " of material to its reserves.";
00952 
00953   // this is a hack! Whatever *was* left in spillover now magically becomes this
00954   // new commodity. We need to do something different (maybe) for recycle.
00955   spillover_->Absorb(mat);
00956 
00957   while (!cyclus::IsNegative(spillover_->quantity() - batch_size())) {
00958     Material::Ptr batch;
00959     // this is a hack to deal with close-to-equal issues between batch size and
00960     // the amount of fuel in spillover
00961     if (spillover_->quantity() >= batch_size()) {
00962       batch = spillover_->ExtractQty(batch_size());
00963     } else {
00964       batch = spillover_->ExtractQty(spillover_->quantity());
00965     }
00966     assert(commod != "");
00967     crctx_.AddRsrc(commod, batch);
00968     reserves_.Push(batch);
00969   }
00970 }

00973 cyclus::BidPortfolio<cyclus::Material>::Ptr BatchReactor::GetBids_(
00974     cyclus::CommodMap<cyclus::Material>::type& commod_requests,
00975     std::string commod,
00976     cyclus::toolkit::ResourceBuff* buffer) {
00977   using cyclus::Bid;
00978   using cyclus::BidPortfolio;
00979   using cyclus::CapacityConstraint;
00980   using cyclus::Composition;
00981   using cyclus::Converter;
00982   using cyclus::Material;
00983   using cyclus::Request;
00984   using cyclus::ResCast;
00985   using cyclus::toolkit::ResourceBuff;
00986 
00987   BidPortfolio<Material>::Ptr port(new BidPortfolio<Material>());
00988 
00989   if (commod_requests.count(commod) > 0 && buffer->quantity() > 0) {
00990     std::vector<Request<Material>*>& requests = commod_requests[commod];
00991 
00992     // get offer composition
00993     Material::Ptr back = ResCast<Material>(buffer->Pop(ResourceBuff::BACK));
00994     Composition::Ptr comp = back->comp();

00998     for (it = requests.begin(); it != requests.end(); ++it) {
00999       Request<Material>* req = *it;
01000       double qty = std::min(req->target()->quantity(), buffer->quantity());
01001       Material::Ptr offer = Material::CreateUntracked(qty, comp);
01002       port->AddBid(req, offer, this);
01003     }
01004 
01005     CapacityConstraint<Material> cc(buffer->quantity());
01006     port->AddConstraint(cc);
01007   }
01008
